docker build .

docker run "image"

docker run -p 3000:3000 "image"    #with console    #run in foreground

docker run -p 3000:3000 -d "image"    #without console    #run in background

docker ps    #list of all running container

docker ps -a    #list of all container

docker stop "container name"

docker --help

docker start "container name"    #restart the container

docker start -a "container name"    #restart the container in attach mode

docker attach "container name"    #see the foreground

docker logs "container name"    #see the logs

docker logs -f "container name"    #see the logs and keep on listening

docker images    #List of images

docker images -a    #List of all images

docker inspect "image id"    #Info about that image

docker image prune    #remove all the unuse images

docker rmi "image id"    #delete image    #can do multiple just separate with space

docker rm "container name"    #delete containter    #can do multiple just separate with space

docker run --rm    #Automatically remove the container when stop    #add the --m list a parameter

docker cp "folder name"/"file name or ." "container name":/"path"    #Copy folder or files in the running containter

docker cp "container name":/"path" "folder name"    #Copy from the container to local

docker run --name "custom name" "image id"    #run a container with a name

docker build -t "name":"tag" .    #name and tag for the image

docker tag "old image name and its tag" "new image name"    #rename the image

docker login    #login your dockerhub

docker logout    #logout your dockerhub

#Inter Active mode
docker run -it "image"    #run the container with interactive

docker run -i "image"    #run the container with interactive

docker run -t "image"    #run the container with interactive

docker start -a -i "container name"    #restart the container with input
